## 题解

```ts
function isValidBST(root: TreeNode | null): boolean {
  let prev = -Infinity

  const dfs = (node: TreeNode | null) => {
    if(!node) {
      return true
    }
    if (!dfs(node.left)) {
      return false
    }
    if (prev >= node.val) {
      return false
    }
    prev = node.val
    if (!dfs(node.right)) {
      return false
    }

    return true
  }

  return dfs(root)
};
```
