const findDuplicate = (nums) => {
  let low = 1
  let high = nums.length - 1
  while (low < high) {
    const mid = (low + high) >> 1
    let count = 0
    for (let i = 0; i < nums.length; i++) {
      if (nums[i] <= mid) {
        count++
      }
    }
    if (count > mid) {
      high = mid
    } else {
      low = mid + 1
    }
  }
  return low
}

const findDuplicate = (nums) => {
  let slow = nums[0], fast = nums[0]

  while (slow && fast) {
    slow = nums[slow]
    fast = nums[nums[fast]]

    if (slow === fast) {
      fast = nums[0]

      while (fast !== slow) {
        fast = nums[fast]
        slow = nums[slow]
      }
      return slow
    }
  }
};
