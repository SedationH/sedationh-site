title: 31. Next Permutation
tags: [DFS]

```go
func nextPermutation(nums []int)  {
  A := -1
  for i := len(nums) - 2; i >= 0; i-- {
    if nums[i] < nums[i + 1] {
      A = i
      break
    }
  }
  if A != -1 {
    for i := len(nums) - 1; i > A; i-- {
      if nums[i] > nums[A] {
        nums[i], nums[A] = nums[A], nums[i]
        break
      }
    }
  }
  for i, j:= A + 1, len(nums) - 1; i < j; i, j = i + 1, j - 1 {
    nums[i], nums[j] = nums[j], nums[i]
  }
}
```
